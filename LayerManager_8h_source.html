46 template <typename WrappedIterator>
47 class DirtyOnlyIterator
48 // : public std::iterator<std::forward_iterator_tag,

53 
57  DirtyOnlyIterator(WrappedIterator it, WrappedIterator end):
58  m_iter(it),
59  m_end(end)
60  {
61  SetToNextDirty();
62  }
63 
66  DirtyOnlyIterator(const DirtyOnlyIterator& other):
67  m_iter(other.m_iter),
68  m_end(other.m_end)
69  {
70  SetToNextDirty();
71  }
72 
75  DirtyOnlyIterator& operator++()
76  {
77  ++m_iter;
78  SetToNextDirty();
79  return *this;
80  }
81 
84  DirtyOnlyIterator operator++(int)
85  {
86  WrappedIterator tmp(*this);
87  operator++();
88  return tmp;
89  }
90 
94  bool operator == (const DirtyOnlyIterator& rhs) const
95  {
96  // You could argue we should check m_end too,
97  // but all we really care about is whether we're pointed
98  // at the same place, and it's faster...
99  return m_iter == rhs.m_iter;
100  }
101 
105  bool operator != (const DirtyOnlyIterator& rhs) const
106  {
107  return m_iter != rhs.m_iter;
108  }
109 
111  value_type& operator * ()
112  { return *m_iter; }
113 
114 private:
115  void SetToNextDirty()
116  {
117  while (m_iter != m_end && !m_iter->first->IsDirty())
118  {
119  ++m_iter;
120  }
121  }
122 
123  WrappedIterator m_iter;
124  WrappedIterator m_end;
125 };

336  // Note that the layers attribute should ONLY used during serialization, as this is the ONLY
337  // times at which these attributes are guaranteed to "line up" to the internal layer register
338  // (m_layerList). Ie, immediately before save (due to the pre-save callback), the attributes will
339  // be written from m_layerList; and immediate after open (due to the post-open callback), m_layerList
340  // will be initialized from the attributes. At all other times, the attributes will be OUT OF SYNC
341  // (and, in fact, are intentionally set to be "empty", so there's no confusion / someone doesn't
342  // try to use "out of date" information)
343  AL_DECL_ATTRIBUTE(layers);
